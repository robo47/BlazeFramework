namespace blaze\web\tag;
use blaze\lang\Object;

/**
 * Description of HtmlTag
 *
 * @author  Christian Beikov
 * @license http://www.opensource.org/licenses/gpl-3.0.html GPL
 * @link    http://blazeframework.sourceforge.net
 * @see     Classes which could be useful for the understanding of this class. e.g. ClassName::methodName
 * @since   1.0
 * @version $Revision$
 * @todo    Something which has to be done, implementation or so
 */
abstract class AbstractTag extends Object implements Tag {

    private $elements = array();
    private $parent;

    public function isExpression($value) {
        $value = \blaze\lang\String::asWrapper($value);
        $index = $value->indexOf('{');
        
        if($index != -1 && $value->indexOf('}') > $index)
                return true;
        return false;
    }


     public function renderSubElements(){
         $output = '';
        foreach($this->elements as $element)
                $output .= $element->render();
        return $output;
     }

     public function add(Tag $elem){
        $this->elements[] = $elem;
        $elem->setParent($this);
        return $this;
     }

    public function setParent($parent){
        $this->parent = $parent;
        return $this;
    }

    public function getParent(){
        return $this->parent;
    }

    public function getChildren(){
        return $this->elements;
    }
    
    public function processChildrenDecodes(\blaze\web\application\BlazeContext $context) {
        foreach($this->elements as $child){
                $child->processDecodes($context);
            }
    }

    public function processDecodes(\blaze\web\application\BlazeContext $context) {
        $this->processChildrenDecodes($context);
    }

    public function processApplication(\blaze\web\application\BlazeContext $context) {
        $this->processChildrenApplication($context);
    }

    public function processChildrenApplication(\blaze\web\application\BlazeContext $context) {
            foreach($this->elements as $child){
                $child->processApplication($context);
            }
    }

    public function processChildrenUpdates(\blaze\web\application\BlazeContext $context) {
        foreach($this->elements as $child){
                $child->processUpdates($context);
            }
    }

    public function processChildrenValidations(\blaze\web\application\BlazeContext $context) {
        foreach($this->elements as $child){
                $child->processValidations($context);
            }
    }

    public function processUpdates(\blaze\web\application\BlazeContext $context) {
        $this->processChildrenUpdates($context);
    }

    public function processValidations(\blaze\web\application\BlazeContext $context) {
        $this->processChildrenValidations($context);
    }



}

?>
